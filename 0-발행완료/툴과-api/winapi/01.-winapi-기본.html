1. WinAPI 기본

Note

과거엔 WinAPI 어플리케이션을 작성할 때 헝가리안 표기법을 사용했다. MSDN에 따르면 현재는 권장되지 않는다. 물론, 지금은 WinAPI로 어플리케이션을 만든다는 게 언어도단이다.

WinAPI란

마이크로소프트에서 제공하는, 윈도우즈 OS를 위한 C언어 기반의 API. 윈도우즈에서 동작하는 모든 프로그램은 결국 WinAPI를 기반으로 동작한다.
예컨대 C++ 개발자들을 위한 MFC나 .NET 기반의 윈도우즈 프레임워크들도 내부적으로 WinAPI를 사용하며, 유니티나 언리얼같은 게임 엔진들도 윈도우즈 플랫폼 빌드시 WinAPI를 사용하게 된다.

Win32API, Windows Classic API라고 부르기도 한다.

창 (Window)

윈도우즈 어플리케이션에서 창은 응용프로그램이 출력을 표시하고 사용자로부터 입력을 수신하는 사각형 영역이다. GUI 어플리케이션 개발의 첫 단계는 메인 창을 만드는 것이다.

사용자의 입력(이벤트의 발생)은 한 번에 하나의 창에서만 일어난다.

클라이언트 영역

창의 내부 영역으로 창에 대한 입력을 처리하고 출력을 표시하는 영역이다.

비클라이언트 영역

제목 표시줄, 메뉴 모음, 창 메뉴, 최소화-최대화 단추, 스크롤 막대 등 기본적으로 생성되는 요소들의 영역을 비클라이언트 영역이라고 한다.

컨트롤

단추, 텍스트 입력창 등 특정한 정보를 출력하거나 사용자로부터 얻는 용도로 사용되는, 다른 창의 내부에 사용되는 창을 말한다. 주의할 것은 컨트롤들도 모두 창의 일종이라는 것이다.

창 클래스 (Window Class)

창 클래스는 창의 기반으로 창의 속성을 정의하고 해당 창에서 발생한 이벤트(전달된 메시지)를 어떻게 처리할 지(프로시저 함수), 이 창이 어떤 인스턴스(프로그램)에 속하는지 정의한다.

하나의 창 클래스에 속한 여러 창이 있을 수 있으며, 창 클래스 단위로 창의 속성이 설정되고 프로시저 함수를 작성하므로 이들은 모두 같은 모양과 동작을 하게 된다.

창 클래스 등록하기

작성한 창 클래스로부터 실제 창을 만들어 메모리에 적재하려면, 우선 창 클래스를 OS가 인식할 수 있는 대상으로 포함해야 한다. 이는 RegisterClassEx 함수가 담당한다.

인스턴스 핸들과 창 핸들

기본적으로 ‘인스턴스’라는 건 현재 실행되고 있는 프로그램을 말한다. 즉 인스턴스 핸들이란 현재 프로그램의 고유한 ID같은 것이다. 타입은 HINSTANCE.
반면 창 핸들은 프로그램 안에서 생성된 창들에 대한 핸들이다. 타입은 HWND.

WinAPI 어플리케이션 기본 구조

라이브러리

WinAPI는 C언어 기반 API이며 C언어의 다음 헤더가 대부분의 WinAPI 문법을 포함한다.

#include <Windows.h>

윈도우 생성하기

윈도우를 생성하는 과정은 다음과 같다.

// 1. 윈도우 클래스를 정의하고 OS에 등록한다.

WNDCLASSEXW wcex; // 창 클래스 구조체

wcex.cbSize = sizeof(WNDCLASSEX);
wcex.style = CS_HREDRAW | CS_VREDRAW; // 여러 속성들을 지정
...
wcex.hInstance = hInstance; // 창이 속하는 인스턴스(현재 프로그램)
wcex.lpfnWndProc = WndProc; // 해당 창에서 이벤트가 발생(메시지가 전달)되었을 때 호출되는 프로시저 함수
wcex.lpszClassName = szWindowClass; // 이 창 클래스의 이름. 이후 창을 생성해 메모리에 적재할 때 이 이름으로 구분한다.

return RegisterClassExW(&wcex); // 정의한 클래스를 등록한다.



// 2. 창을 생성한다.

HWND hWnd = CreateWindowW( // 창을 생성한다. 메모리에 적재한다는 뜻이다.
				szWindowClass, // 생성할 창의 기반 클래스의 이름
				szTitle, // 창의 이름. 메인 창이라면 타이틀로도 쓰인다.
				WS_OVERLAPPEDWINDOW, // 여러 속성을 지정한다.
				...);

ShowWindow(hWnd, nCmdShow); // 창을 표시한다.
UpdateWindow(hWnd); // 창을 업데이트한다.

윈도우 프로시저 함수

짝지어진 윈도우에서 발생한 이벤트를 처리하는 함수. 창 클래스를 정의할 때 프로시저 함수 포인터를 넣어서 연결한 바 있다. (이를 Hook이라고 표현.)

함수의 내부 구조는 다음과 같다.

LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
{
	switch (message) // 일반적으로 switch 문으로 message의 종류를 구분한다.
	{
	case WM_COMMAND:
		...
	...
	default:
		return DefWindowProc(hWnd, message, wParam, lParam); // DefWindowProc 함수에는 메시지 종류별로 기본 동작이 정의되어 있다. 특별히 동작을 지정해줄 메시지만 위에서 정의해주고, 나머지는 DefWindowProc에 맡기면 된다.
	}
	
	return 0;
}

보면 알겠지만 hWnd에 이벤트가 발생한 윈도우의 핸들이, message에 발생한 이벤트의 종류가, wParam과 lParam에 해당 이벤트의 처리를 위한 부가 정보가 담겨서 호출된다.

메시지의 종류는 무수히 많으므로 그 때 그 때 찾아보자.

WinMain 함수

WinMain 함수는 C의 main 함수의 래퍼로, WinAPI 어플리케이션의 진입점이다.

메시지 루프

사용자가 여러가지 메시지(이벤트)를 발생시키면 WinAPI 어플리케이션은 이를 메시지 큐에 담아놓고, 주기적으로 루프에 담긴 메시지를 꺼내 윈도우 프로시저로 전달하는 방식으로 이를 처리한다.

MSG msg; // 메시지를 담을 변수
while (GetMessage(&msg, nullptr, 0, 0)) // GetMessage로 메시지 큐에서 메시지를 꺼낸다
{
	TranslateMessage(&msg); // 문자 입력 시 필요하다.
	DispatchMessage(&msg); // 꺼낸 메시지를 적당한 윈도우 프로시저에 전달하며 호출하는 역할.
}

GetMessage 함수는 메시지 큐가 비었으면 새 메시지가 담길 때까지 대기하며 FALSE를 반환할 때는 종료 메시지가 담겼을 때 뿐이다.
이와 달리 PeekMessage는 메시지가 있으면 꺼낸 뒤 TRUE를, 없으면 FALSE를 반환하고 넘어간다.
게임에서는 플레이어가 조작을 하지 않아도 프레임을 진행해야 하므로 PeekMessage를 사용한다.

TranslateMessage 함수는 키보드 입력 이벤트가 발생했을 때 이를 꺼내보고 엄밀한 의미를 가지도록 가공한 메시지(WM_CHAR)로 만들어 다시 메시지 큐에 넣는 역할을 한다.
즉, 키보드 키를 누르게 되면 WM_KEYDOWN 발생 → TranslateMessage가 꺼내보고 WM_CHAR 메시지로 가공하여 메시지 큐에 삽입 → DispatchMessage가 그걸 꺼내보고 프로시저를 호출하는 방식이다.
꼭 필요한 함수는 아니지만, WM_KEYDOWN 메시지와 함께 전달되는 정보는 바로 사용하기 곤란한 형태이다보니 문자 입력을 받을 일이 있다면 TranslateMessage()를 메시지 루프에 추가하고 WM_CHAR 메시지로 다뤄야 한다.

메시지

모든 메시지는 특정한 이벤트와 매핑되어있다. 이벤트가 발생하면 OS가 알맞은 메시지를 메시지 큐에 담게 된다.

메시지의 부가 정보는 용도에 따라 WPARAM, LPARAM 두 타입의 변수에 담겨지는데, WPARAM은 Word Parameter(2바이트), LPARAM은 Long Parameter(4바이트)라는 의미로 정보의 크기에 따라 분리한 것이었으나, Win32 API에선 Word도 32비트(4바이트)가 되어 둘의 크기는 똑같고 담기는 정보의 성격만 달라졌다.

메시지의 정보를 추출하는 법

예컨대 마우스 클릭 이벤트가 발생하면 LPARAM 매개변수에 클릭한 좌표가 담겨 전달된다. 이 때 Y 좌표는 LPARAM의 4바이트 중 앞 2바이트에, X 좌표는 뒤 2바이트에 담기는데, 이걸 추출해낼 때는 WinAPI가 정의해둔 매크로를 사용하면 된다.

case WM_LBUTTONDOWN:
	int mouseY = HIWORD(lParam); // HIWORD 매크로 함수는 4바이트의 앞 2바이트를 반환한다.
	int mouseX = LOWORD(lParam); // LOWORD 매크로 함수는 4바이트의 뒤 2바이트를 반환한다.

코드 개괄

일부 불필요한 코드와 지금은 생략할 부분을 제외한 기본 구조는 다음과 같다.

HINSTANCE hInst; // 프로그램의 인스턴스 핸들을 전역적으로 저장한다.
HWND hWnd; // 필요하다면 메인 윈도우 핸들도 전역적으로 저장하곤 한다.

WCHAR szTitle[100]; // 메인 윈도우의 이름(타이틀 바에 출력할 텍스트)
WCHAR szWindowClass[100]; // 메인 윈도우 클래스의 이름

int APIENTRY WinMain(HINSTANCE hInstance, ...)
{
	...
	MyRegisterClass(hInstance); // 메인 윈도우 클래스를 등록한다.
	
	if (!InitInstance(hInstance, nCmdShow)) // 메인 윈도우를 생성하고 표시한다.
	{
		return FALSE;
	}
	
	// 메인 메시지 루프
	MSG msg;
	while (GetMessage(&msg, nullptr, 0, 0))
	{
		TranslateMessage(&msg);
		DispatchMessage(&msg);
	}
}

ATOM MyRegisterClass(HINSTANCE hInstance)
{
	WNDCLASSEXW wcex; // 접사 w는 언제나 wide 문자열을 지원한다는 뜻
	
	wcex.style = CS_HREDRAW | CS_VREDRAW;
	...
	wcex.hInstance = hInstance;
	wcex.lpfnWndProc = WndProc;
	wcex.lpszClassName = szWindowClass;
	
	return RegisterClassExW(&wcex); // 먼저 윈도우 클래스를 등록한다.
}

BOOL InitInstance(HINSTANCE hInstance, int nCmdShow)
{
	HWND hWnd = CreateWindowW(szWindowClass, szTitle, ...); // 윈도우 인스턴스를 생성한다.

	if (!hWnd)
	{
		return FALSE;
	}

	ShowWindow(hWnd, nCmdShow);
	UpdateWindow(hWnd);
	
	return TRUE;
}

LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
{
	switch (message)
	{
	case WM_COMMAND: // 대표적인 이벤트1. 사용자가 특정 커맨드를 발동시킨 경우
		{
			int wmId = LOWORD(wParam); // 커맨드의 아이디가 wParam에 담겨서 온다.
			switch (wmId)
			{
			case IDM_ABOUT: // 메뉴에서 정보를 누른 경우
				DialogBox(hInst, ..., About); // 다이얼로그박스는 기본 지원하는 작은 창이다. 마지막 매개변수로 지금 열 다이얼로그 박스 윈도우의 프로시저 함수를 지정한다.
				break;
			case IDM_EXIT: // 종료 버튼을 누른 경우
				DestroyWindow(hWnd); // 윈도우를 메모리에서 제거한다. 제거되기 전에 WM_DESTROY 메시지가 전달된다.
				break;
			default:
				return DefWindowProc(hWnd, message, wParam, lParam);
			}
		}
		break;
	case WM_PAINT: // 대표적인 이벤트2. 그리기 명령. 다시 그려야 하는 영역이 생길 때 메시지가 발생한다.
		{
			// 이하의 GDI, GDI+ 내용은 나중에 배우겠다.
			// 대충 창 전체를 COLOR_WINDOW + 2 색으로 채우는 사각형을 그리는 코드
			PAINTSTRUCT ps;
			HDC hdc = BeginPaint(hWnd, &ps);
			FillRect(hdc, &ps.rcPaint, (HBRUSH)(COLOR_WINDOW + 2));
			EndPaint(hWnd, &ps);
		}
		break;
	case WM_DESTROY: // 윈도우가 메모리에서 제거되기 전에 전달되는 메시지
		PostQuitMessage(0); // 메시지 큐에 메시지 루프를 종료하라는 마지막 메시지를 담아 메시지 루프를 종료시킨다.
		break;
	default:
		return DefWindowProc(hWnd, message, wParam, lParam);
	}
	
	return 0;
}

// 앞서 다이얼로그 윈도우를 만들면서 전달해줬던 프로시저 함수
// 이렇게 메인 윈도우 외의 다른 일반 윈도우들의 프로시저 함수도 메인 윈도우의 프로시저 함수와 똑같다.
INT_PTR CALLBACK About(HWND hDlg, UINT message, WPARAM wParam, LPARAM lParam)
{
  UNREFERENCED_PARAMETER(lParam);
  switch (message)
  {
  case WM_INITDIALOG:
    return (INT_PTR)TRUE;

  case WM_COMMAND:
    if (LOWORD(wParam) == IDOK || LOWORD(wParam) == IDCANCEL)
    {
      EndDialog(hDlg, LOWORD(wParam));
      return (INT_PTR)TRUE;
    }
    break;
  }
  return (INT_PTR)FALSE;
}
